# Verifica se o nó "cabeca" existe antes de tentar usá-lo
	if cabeca:
		cabeca.look_at(get_global_mouse_position())
		
		
		
func virar_para_o_mouse():
	# Verifica a posição do mouse e inverte a escala do sprite
	var mouse_pos = get_global_mouse_position()
	if mouse_pos.x < global_position.x:
		# Vira o sprite para a esquerda
		self.scale.x = -1  # Inverte o eixo X, não a escala global inteira
	else:
		# Vira o sprite para a direita
		self.scale.x = 1

func _physics_process(delta):
	girar_objeto()
	virar_para_o_mouse()
