extends StaticBody2D

onready var trampolim_area = $area_moralista

var FORCA_PULO_TRAMPOLIM = -2000
var a1_trampolim = false


func _ready():
	trampolim_area.connect("body_exited", self, "_on_area_body_exited")
		
func _on_area_body_exited(body):
	if body.is_in_group("trampolim_pulavel"):
		a1_trampolim = false
		
func trampolim_audios():
	if a1_trampolim == true:
		$audio_trump_esq/audio_esquerdo.play()
		
func efeito_trampolim():
	for body in trampolim_area.get_overlapping_bodies():
		if body.is_in_group("trampolim_pulavel"):
			a1_trampolim = true
			variavel_global.j_frente = -1
			variavel_global.j_direcao = -1
			
			if "velocidade" in body:
				body.velocidade.x = FORCA_PULO_TRAMPOLIM
				
			if "trampolim_horizontal" in body:
				body.trampolim_horizontal = true
				
		
func _process(_delta):
	trampolim_audios()

func _physics_process(_delta):
	efeito_trampolim()
